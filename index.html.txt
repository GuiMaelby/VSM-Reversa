        function resetCardStyles() {
            const cards = document.querySelectorAll('.step-card');
            cards.forEach(card => card.classList.remove('active'));
        }

        function showDetails(stageKey) {
            const data = vsmData[stageKey];
            const container = document.getElementById('dynamic-content');
            const flowContainer = document.getElementById('det-flow');
            
            resetCardStyles();
            document.getElementById(`card-${stageKey}`).classList.add('active');

            document.getElementById('placeholder-text').classList.add('hidden');
            container.classList.remove('hidden');

            document.getElementById('det-tag').innerText = data.tag;
            document.getElementById('det-title').innerText = data.title;
            document.getElementById('det-status').innerText = data.status;

            flowContainer.innerHTML = ''; 

            data.flow.forEach(item => {
                if (item.type === 'info') {
                    const div = document.createElement('div');
                    div.className = item.class;
                    div.innerHTML = item.text;
                    flowContainer.appendChild(div);
                } 
                else if (item.type === 'card') {
                    const card = document.createElement('div');
                    card.className = "flex items-start gap-4 p-4 bg-claro-gray-light rounded-lg border border-claro-gray-medium transition-colors hover:bg-white";
                    card.innerHTML = `
                        <div class="text-2xl p-2 bg-white rounded-lg border border-claro-gray-medium shrink-0">${item.icon}</div>
                        <div class="grid grid-cols-1 md:grid-cols-3 gap-x-4 gap-y-1 w-full">
                            <div>
                                <span class="text-xs font-semibold text-claro-gray-dark/70 uppercase tracking-wider block">Sub-etapa</span>
